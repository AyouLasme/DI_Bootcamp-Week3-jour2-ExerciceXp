    <body> 
        <p>Input radius value and get the volume of a sphere.</p> 
        <form  id="MyForm"> 
            <label for="radius">Radius</label><input type="text" name="radius" id="radius" required> 
            <label for="volume">Volume</label><input type="text" name="volume" id="volume"> 
            <input type="submit" value="Calculate" id="submit">    
        </form> 


        <script>
            // obetenir le form et les elements de input
            const form = document.querySelector('#MyForm');
            const radiusInput = document.querySelector('#radius');
            const volumeInput = document.querySelector('#volume');

            // Ajouter un element submit
            form.addEventListener('submit', (event) => {
                // empecher le comportement par defaut
                event.preventDefault();

                // Obtenir la valeur radius
                const radius = radiusInput.value;

                // Calculer le volume de la sphere
                const volume = (4 / 3) * Math.PI * Math.pow(radius, 3);

                // afficher le volume dans le volume de l'element input
                volumeInput.value = volume;
            });
        </script>
